<script>
  const uploadInput = document.getElementById('upload');
  const outputDiv = document.getElementById('output');
  const downloadBtn = document.getElementById('downloadBtn');

  // Simpele achtergrondverwijderaar demo (fake: maakt witte achtergrond transparant)
  // Voor echte AI-achtergrondverwijdering kun je later een API koppelen

  function removeBackground(image) {
    return new Promise((resolve) => {
      const canvas = document.createElement('canvas');
      const ctx = canvas.getContext('2d');
      canvas.width = image.width;
      canvas.height = image.height;
      ctx.drawImage(image, 0, 0);

      const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
      const data = imageData.data;

      // Maak witte pixels transparant
      for (let i = 0; i < data.length; i += 4) {
        const r = data[i];
        const g = data[i + 1];
        const b = data[i + 2];
        if (r > 240 && g > 240 && b > 240) {
          data[i + 3] = 0;
        }
      }
      ctx.putImageData(imageData, 0, 0);

      resolve(canvas);
    });
  }

  function cropToSquare(canvas) {
    const size = Math.min(canvas.width, canvas.height);
    const croppedCanvas = document.createElement('canvas');
    croppedCanvas.width = 3000;  // Vinted max
    croppedCanvas.height = 3000;
    const ctx = croppedCanvas.getContext('2d');

    // Bereken crop startpunt
    const startX = (canvas.width - size) / 2;
    const startY = (canvas.height - size) / 2;

    ctx.drawImage(canvas, startX, startY, size, size, 0, 0, 3000, 3000);

    return croppedCanvas;
  }

  function addLogo(canvas) {
    const ctx = canvas.getContext('2d');
    ctx.font = 'bold 150px serif';
    ctx.fillStyle = 'rgba(212,175,55, 0.6)'; // goud met transparantie
    ctx.textBaseline = 'bottom';
    ctx.fillText('L&W', canvas.width - 400, canvas.height - 50);
  }

  uploadInput.addEventListener('change', async (e) => {
    const file = e.target.files[0];
    if (!file) return;

    const img = new Image();
    img.onload = async () => {
      let canvas = await removeBackground(img);
      canvas = cropToSquare(canvas);
      addLogo(canvas);

      outputDiv.innerHTML = '';
      outputDiv.appendChild(canvas);

      // Download knop aanmaken
      downloadBtn.style.display = 'inline-block';
      downloadBtn.onclick = () => {
        const link = document.createElement('a');
        link.download = 'vinted-cleaned.png';
        link.href = canvas.toDataURL('image/png');
        link.click();
      };
    };
    img.src = URL.createObjectURL(file);
  });
</script>
